            void floyds(int wt[100][100], int n, int D[100][100]) {
                for (int i = 0; i < n; i++) {
                    for (int j = 0; j < n; j++) {
                        D[i][j] = wt[i][j];
                    }
                }
            
                for (int k = 0; k < n; k++) {
                    for (int i = 0; i < n; i++) {
                        for (int j = 0; j < n; j++) {
                            if (D[i][j] > D[i][k] + D[k][j]) {
                                D[i][j] = D[i][k] + D[k][j];
                            }
                        }
                    }
                }
                
                printf("Shortest path matrix\n");
                
                 for (int i = 0; i < n; i++) {
                    for (int j = 0; j < n; j++) {
                        printf("%d ",D[i][j]);
                     
                    }
                    printf("\n");
                }
            }
            
            int main() {
                int wt[100][100], D[100][100];
                int n;
            
                printf("Enter the number of vertices: ");
                scanf("%d", &n);
            
                printf("Enter the weight matrix (enter 0 for no direct path):\n");
                for (int i = 0; i < n; i++) {
                    for (int j = 0; j < n; j++) {
                        scanf("%d", &wt[i][j]);
                        if (i!=j && wt[i][j] == 0) {  
                            wt[i][j] = 99999; 
                        }
                    }
                }
            
                floyds(wt, n, D);
            
              
            
                return 0;
            }

void warshals(int adj[100][100], int n, int R[100][100]) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            R[i][j] = adj[i][j];
        }
    }

    for (int k = 0; k < n; k++) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
            R[i][j]=R[i][j]||R[i][k]&&R[k][j];
                
            }
        }
    }
    
    printf("Transitice closure matrix\n");
    
     for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
        	printf("%d ",R[i][j]);
         
        }
        printf("\n");
    }
}
