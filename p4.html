Kruskal's Algorithm Implementation


#include <stdio.h>
#include <stdlib.h>

int i, j, a, b, u, v, n, ne = 1;
int min, mincost = 0, cost[9][9], parent[9];

int find(int i) {
    while (parent[i])
        i = parent[i];
    return i; // return the root
}

int uni(int i, int j) { // performs the union of 2 sets
    if (i != j) { // if roots are different then only perform union
        parent[j] = i; // make the common root for both the sets
        return 1; // success
    }
    return 0; // else failure
}

void main() {
    printf("\n\n\tImplementation of Kruskal's algorithm\n\n");
    printf("\nEnter the no. of vertices\n");
    scanf("%d", &n);
    printf("\nEnter the cost adjacency matrix\n");
    for (i = 1; i <= n; i++) {
        for (j = 1; j <= n; j++) {
            scanf("%d", &cost[i][j]);
            if (cost[i][j] == 0)
                cost[i][j] = 999;
        }
    }
    printf("\nThe edges of Minimum Cost Spanning Tree are\n\n");
    while (ne < n) { // each spanning tree has n-1 edges so this will run n-1 times
        for (i = 1, min = 999; i <= n; i++) {
            for (j = 1; j <= n; j++) {
                if (cost[i][j] < min) {
                    min = cost[i][j];
                    a = u = i;
                    b = v = j;
                }
            }
        }

        u = find(u);
        v = find(v);
        if (uni(u, v)) {
            printf("\n%d edge (%d,%d) = %d\n", ne++, a, b, min);
            mincost += min;
        }
        cost[a][b] = cost[b][a] = 999; // set the vertex as visited
    }
    printf("\n\tMinimum cost = %d\n", mincost);
}
