                void primsALG(int G[100][100], int n) {
                    int v1, v2,min_dist, total=0;//min distance for each iteration.
                    int tree[100];
                
                    //initially no edge is included in the mst
                    for (int i = 0; i < n; i++) {
                        tree[i] = 0;
                    }
                    tree[0] = 1; // Start with the first vertex
                
                    for (int k = 0; k < n - 1; k++) {//runs n-1 time since the mst has n-1 edges for n vertices
                        min_dist = 9999;
                
                        for (int i = 0; i < n; i++) {
                            if (tree[i]) {// if the vertex is present in the mst 
                                for (int j = 0; j < n; j++) {//find an edge which is not present in the mst with the min-dist
                                    if (!tree[j] && G[i][j] && G[i][j] < min_dist) {
                                        min_dist = G[i][j];
                                        v1 = i;
                                        v2 = j;
                                    }
                                }
                            }
                        }
                
                        printf("\n%d --> %d: min_dist = %d", v1, v2, min_dist);
                        total += min_dist;
                        tree[v2] = 1;
                    }
                
                    printf("\nThe total path length is = %d\n", total);
                }
                
                int main() {
                    int G[100][100];
                    int n;
                
                    printf("Enter the number of vertices: ");
                    scanf("%d", &n);
                
                    printf("Enter the adjacency matrix (enter 0 for no direct path):\n");
                    for (int i = 0; i < n; i++) {
                        for (int j = 0; j < n; j++) {
                            scanf("%d", &G[i][j]);
                            if (G[i][j] == 0) {
                                G[i][j] = 9999; 
                            }
                        }
                    }
                
                    primsALG(G, n);
                
                    return 0;
                }
